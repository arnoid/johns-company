<script type="module">
  import mermaid from 'https://cdn.jsdelivr.net/npm/mermaid@10/dist/mermaid.esm.min.mjs';
  mermaid.initialize({ startOnLoad: true });
  
  document.addEventListener("DOMContentLoaded", function() {
    // Find all markdown code blocks generated by Jekyll with the 'language-mermaid' class
    const mermaidElements = document.querySelectorAll('code.language-mermaid');
    
    mermaidElements.forEach(el => {
      // Create a div wrapper for the mermaid diagram
      const newDiv = document.createElement('div');
      newDiv.className = 'mermaid';
      newDiv.textContent = el.textContent;
      
      // Jekyll usually wraps code blocks in <pre> tags, replace the whole <pre>
      const parent = el.parentElement;
      if (parent && parent.tagName.toLowerCase() === 'pre') {
        parent.replaceWith(newDiv);
      } else {
        el.replaceWith(newDiv);
      }
    });
  });
</script>
